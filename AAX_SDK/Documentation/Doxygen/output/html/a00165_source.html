545 template <typename T>
546 const AAX_CString& AAX_CParameter<T>::ShortenedName(int32_t iNumCharacters) const
547 {
548  //More characters than the full name of no specific shortened names.
549  if ((iNumCharacters >= int32_t(mName.Length())) || (mShortenedNames.empty()) || (0 > iNumCharacters))
550  return mName;
551 
552  std::map<uint32_t, AAX_CString>::const_iterator iter = mShortenedNames.upper_bound(static_cast<uint32_t>(iNumCharacters));
553 
554  //If the iterator is already pointing to shortest name, return that.
555  if (iter == mShortenedNames.begin())
556  return iter->second;
557 
558  //lower_bound() will return the iterator that is larger than the desired value, so decrement the iterator.
559  --iter;
560  return iter->second;
561 }

571 template<typename T>
572 void AAX_CParameter<T>::SetValue( T newValue )
573 {
574  double newNormalizedValue = mTaperDelegate->RealToNormalized(newValue);
575 
576  // <DMT> Always go through the automation delegate even if the control isn't automatable to prevent fighting with other GUIs.
577  // Somewhere back in the automation delegate, or elsewhere in the system, it will determine the differences in behavior surrounding
578  // automation. The only reason that there wouldn't be an automation delegate is if this parameter has yet to be added to a
579  // ParameterManager. Let's put the null value guards in place, just in case, and also for unit tests.
580  if ( mAutomationDelegate )
581  {
582  //TODO: Create RAII utility class for touch/release
583 
584  //Touch the control
585  Touch();
586 
587  //Send that token.
588  mAutomationDelegate->PostSetValueRequest(Identifier(), newNormalizedValue );
589 
590  //Release the control
591  Release();
592  }
593  else
594  {
595  mNeedNotify = true;
596 
597  // In the rare case that an automation delegate doesn't exist, lets still set the value. It's possible that someone is trying to
598  // set the new value before adding the parameter to a parametermanager.
599  UpdateNormalizedValue(newNormalizedValue);
600  }
601 }

603 template <typename T>
604 void AAX_CParameter<T>::UpdateNormalizedValue(double newNormalizedValue)
605 {
606  T newValue = mTaperDelegate->NormalizedToReal(newNormalizedValue);
607  if (mNeedNotify || (mValue.Get() != newValue))
608  {
609  //Set the new value
610  mValue.Set(newValue);
611 
612  //<DMT> Always notify that the value has changed through the automation delegate to guarantee that all control surfaces and other
613  // GUIs get their values updated.
614  if (mAutomationDelegate)
615  mAutomationDelegate->PostCurrentValue(Identifier(), newNormalizedValue);
616 
617  // clear flag
618  mNeedNotify = false;
619  }
620 }

622 template <typename T>
623 void AAX_CParameter<T>::InitializeNumberOfSteps()
624 {
625  if (mNumSteps == 0) // If no explicit number of steps has been set...
626  {
627  switch (mControlType)
628  {
629  case AAX_eParameterType_Discrete:
630  {
631  // Discrete parameters default to binary unless
632  // otherwise specified
633  SetNumberOfSteps (eParameterDefaultNumStepsDiscrete);
634  break;
635  }
636  case AAX_eParameterType_Continuous:
637  {
638  // Defaulting to 128 steps to match one full rotation of
639  // Command|8 and similar surfaces, which query the num
640  // steps to determine tick values for rotary encoders
641  SetNumberOfSteps (eParameterDefaultNumStepsContinuous);
642  break;
643  }
644  default:
645  {
646  AAX_ASSERT (0); // Invalid type
647  break;
648  }
649  }
650  }
651 }

793 template<typename T>
794 uint32_t AAX_CParameter<T>::GetStepValue() const
795 {
796  double numSteps = (double) this->GetNumberOfSteps ();
797  if ( numSteps < 2.0 )
798  return 0;
799 
800  double value = this->GetNormalizedValue ();
801  double valuePerStep = 1.0 / ( numSteps - 1.0 );
802  double curStep = ( value / valuePerStep ) + 0.5;
803  if ( curStep < 0.0 )
804  curStep = 0.0;
805  else if ( curStep > (double) ( numSteps - 1.0 ) )
806  curStep = (double) ( numSteps - 1.0 );
807 
808  return (uint32_t) curStep;
809 }

812 void AAX_CParameter<T>::SetStepValue(uint32_t iStep)
813 {
814  double numSteps = (double) this->GetNumberOfSteps ();
815  if ( numSteps < 2.0 )
816  return;
817 
818  double valuePerStep = 1.0 / ( numSteps - 1.0 );
819  double value = valuePerStep * (double) iStep;
820  if ( value < 0.0 )
821  value = 0.0;
822  else if ( value > 1.0 )
823  value = 1.0;
824 
825  this->SetNormalizedValue ( value );
