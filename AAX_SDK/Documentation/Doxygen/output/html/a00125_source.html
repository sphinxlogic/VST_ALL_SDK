281 {
282  return static_cast<AAX_CBoolean>(
283  (44100.0 == inSR) ? ((iMask & AAX_eSampleRateMask_44100) != 0) :
284  (48000.0 == inSR) ? ((iMask & AAX_eSampleRateMask_48000) != 0) :
